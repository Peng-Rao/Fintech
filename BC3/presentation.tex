\begin{frame}{Problem and data}
\textbf{Goal.} Replicate a hidden ``Monster Index'' using only liquid Bloomberg futures, subject to a regulatory VaR cap.
\[
r^{\text{target}}_t = 0.50\,r^{\text{HFRX}}_t + 0.25\,r^{\text{MSCI World}}_t + 0.25\,r^{\text{Global Bond}}_t
\]

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Universe (11 futures).}
\begin{itemize}
  \item Rates: RX1, TY1, DU1, TU2
  \item Equity: ES1, NQ1, VG1, TP1, LLL1
  \item Commodity: GC1, CO1
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Constraints.}
\begin{itemize}
  \item Sample: weekly, Oct 2007 -- Apr 2021 ($\approx 705$ obs).
  \item Gaussian VaR cap: $-\Phi^{-1}(0.01) \cdot \hat\sigma_w \cdot \sqrt{4} \le 8\%$.
  \item Gross-exposure cap: $\sum_j \lvert w_j \rvert \le L$.
  \item Transaction cost: $\tau = 5$ bps on L1 turnover.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Part II --- backtest methodology}
\textbf{Rolling walk-forward.} At every rebalance week:
\begin{itemize}
  \item Fit on the trailing 104-week window.
  \item Predict $w_t$; hold weights until the next rebalance.
  \item No look-ahead --- the training slice ends strictly before evaluation.
\end{itemize}

\smallskip
\textbf{Contract metrics.}
\[
\mathrm{TE} = \sqrt{52\,\mathrm{Var}(r^{\text{rep}}_t - y_t)}, \qquad
\mathrm{IR} = \frac{52\,\mathbb{E}[r^{\text{rep}}_t - y_t]}{\mathrm{TE}}, \qquad
\bar T = \mathbb{E}_t\!\Bigl[\textstyle\sum_j \lvert \Delta w_{t,j} \rvert\Bigr]
\]

\smallskip
\textbf{Transaction-cost subtraction.} Costs hit on rebalance dates only:
\[
c_t = \tau \cdot \sum_{j=1}^{n} \lvert w_{t,j} - w_{t-1,j} \rvert, \qquad
\mathrm{net\_IR} = \frac{52\,\mathbb{E}[r^{\text{rep}}_t - c_t - y_t]}{\mathrm{net\_TE}}
\]

\smallskip
\textbf{Risk projection layer.} Gross-exposure cap first ($\sum|w_j| \le L$), then iterative historical-VaR shrinkage until $\mathrm{VaR}(w_t) \le 8\%$.
\end{frame}

% =============================================================
\begin{frame}{Part III --- Ridge control + constraint sweep}
\textbf{Canonical control baseline.} Ridge with $\alpha = 0.01$, $\Delta t = 4$ weeks. Frictionless vs.\ net-of-cost comparison fixes the headline cost drag.

\smallskip
\textbf{Cadence $\times$ leverage sweep.} Six base configurations plus a 6-axis Cartesian grid:
\[
\Delta t \in \{1, 2, 4, 8, 12\},\quad L \in \{1.0, 1.5, 2.0\},\quad \alpha \in \{10^{-4},\,10^{-3},\,10^{-2}\},\quad \rho_{\text{EN}} \in \{0.2, 0.5, 0.8\}
\]

\smallskip
\textbf{Constraint layer for every configuration:}
\begin{itemize}
  \item Fit MinMax-scaled ElasticNet on the trailing 104-week panel.
  \item Project: $w \mapsto \Pi_{\mathcal{W}}(w)$ with $\mathcal{W} = \{w : \sum_j |w_j| \le L\}$.
  \item Shrink iteratively until $\mathrm{VaR}_{\text{hist}}(w) \le 8\%$.
\end{itemize}

\smallskip
\textbf{Composite ranking.} Z-score $\{\mathrm{IR},\, \mathrm{net\_IR},\, \rho,\, -\mathrm{TE},\, -\bar T,\, -\mathrm{MDD}\}$, average, sort.
\end{frame}

% =============================================================
\begin{frame}{Part IV --- linear benchmark family}
\textbf{Predict-then-optimize.} Each rebalance:
\begin{enumerate}
  \item \textbf{Alpha.} Fit a classical linear model on $(X_{\text{tr}}, y_{\text{tr}})$; extract $\mu = \mathrm{coef\_}$.
  \item \textbf{Risk.} Ledoit--Wolf shrunken sample covariance $\Sigma$ (annualised).
  \item \textbf{Optimize.} Convex problem (CVXPY):
\end{enumerate}
\vspace{-0.4em}
\[
\max_{w} \;\; \mu^{\top} w - \tfrac{1}{2}\lambda\, w^{\top}\Sigma w - \tau\,\lVert w - w_{t-1}\rVert_1
\quad \text{s.t.} \quad \lVert w \rVert_1 \le L,\;\; \lvert w_j \rvert \le 0.5
\]
\vspace{-0.5em}

\smallskip
\textbf{Leaderboard sweep.}
\[
\bigl\{\text{OLS},\, \text{Ridge}_{\alpha},\, \text{Lasso}_{\alpha},\, \text{ElasticNet}_{(\alpha, \rho)},\, \text{Huber}_{\delta}\bigr\} \times \bigl\{\text{all-11},\, \text{top-5}\bigr\}
\]

\smallskip
\textbf{Top-$K$ pre-selection.} At each rebalance, keep the $K$ futures with highest $\lvert \mathrm{corr}(r_j, y) \rvert$ \emph{computed inside the training window} --- no leakage.

\smallskip
\textbf{Sanity check.} With $\lambda = 2$, slack constraints, and no Ledoit--Wolf shrinkage, the convex layer recovers the OLS fit to within solver tolerance.
\end{frame}

% =============================================================
\begin{frame}{Part V --- Kalman filter / state-space}
\textbf{State-space model.} Random-walk weights, scalar return as observation:
\[
\begin{aligned}
x_t &= x_{t-1} + B u_t, & u_t &\sim \mathcal{N}(0, I), &\quad B &= \sigma_w I \\
y_t &= r_t^{\top} x_t + D \varepsilon_t, & \varepsilon_t &\sim \mathcal{N}(0, 1), &\quad D &= \hat\sigma_y
\end{aligned}
\]

\smallskip
\textbf{Three variants compared.}
\begin{itemize}
  \item \textbf{Static $\sigma_w$ grid.} $\sigma_w \in \{10^{-4},\, 10^{-3},\, 10^{-2}\}$; warm-started from Ridge on the first 52 weeks.
  \item \textbf{EM-tuned noise.} \texttt{pykalman.KalmanFilter.em()} fits $(B, D)$ jointly, capped at 3 iterations (the $1$-obs / $11$-state system is under-identified at higher $n$).
  \item \textbf{Regime-switching.} 2-state Gaussian HMM on rolling realised target volatility selects $\sigma_w \in \{\sigma_w^{\text{calm}},\, \sigma_w^{\text{stressed}}\}$.
\end{itemize}

\smallskip
\textbf{Leakage discipline.} The HMM is fit on the 104-week warm-up window only and then \emph{frozen}; subsequent regime labels are produced by \texttt{.predict}, not refit. Without this, the HMM sees post-train data.

\smallskip
\textbf{Risk projection.} Same Gaussian VaR cap $\mathrm{VaR}(\hat x_t) \le 8\%$ applied after every filter update.
\end{frame}

% =============================================================
\begin{frame}{Part VI --- deep learning weight generator}
\textbf{End-to-end framing.} The network outputs the \emph{weights}, not the returns:
\[
w_t = f_\theta(\phi_t), \qquad r^{\text{rep}}_t = w_t^{\top} r_t.
\]

\textbf{Features $\phi_t$} (lagged by one step via \texttt{shift(1)} --- no contemporaneous information):
\begin{itemize}
  \item Trailing compounded returns at $k \in \{4, 12, 52\}$ weeks per asset.
  \item Trailing realised volatility at $k \in \{12, 52\}$ weeks.
  \item Regime indicators: 12-week MSCI World return, 12-week bond trend, 12-week target volatility.
  \item Warm start: previous Ridge weight from a 52-week trailing fit ($\alpha = 10^{-3}$).
\end{itemize}

\textbf{Architecture.} \texttt{WeightMLP}: $\phi_t \in \mathbb{R}^{d_\phi} \to 64 \to 32 \to w_t \in \mathbb{R}^{11}$, ReLU + dropout $0.1$, optional gross-exposure projection inside the forward pass.
\end{frame}

% =============================================================
\begin{frame}{Part VI --- loss, training, rolling backtest}
\textbf{Loss --- forward-window tracking-error variance + turnover penalty:}
\[
\mathcal{L}(\theta) = \mathrm{Var}_{s \in [t,\, t+H-1]}\!\bigl(w_t^{\top} r_s - y_s\bigr) + \lambda\,\bigl\lVert \Delta w_t \bigr\rVert_1, \qquad H = 12,\;\; \lambda \in \{0, 10^{-3}, 10^{-2}\}
\]

\smallskip
\textbf{Training regime.}
\begin{itemize}
  \item Chronological 60 / 20 / 20 split --- no shuffling across the boundary.
  \item Adam, lr $10^{-3}$, weight decay $10^{-5}$, batch $64$, max $500$ epochs, patience $30$.
  \item Early stopping on annualised validation TE.
\end{itemize}

\smallskip
\textbf{Rolling out-of-sample backtest.} \texttt{run\_nn\_rolling\_backtest} retrains on the trailing 156-week window at each rebalance $\Delta t \in \{1, 4\}$, predicts $w_t$, applies the Gaussian VaR projection, and warm-starts the next fit from the previous state dict.

\smallskip
\textbf{Why variance, not MSE?} A constant tracking bias washes out under VaR rescaling; the variance directly targets the IR denominator.
\end{frame}

\begin{frame}{Reading the leaderboard}
\textbf{What the analysis shows.}
\begin{itemize}
  \item \textbf{Cadence sets the cost/fit trade-off.} Weekly rebalancing maximises tracking quality but pays the heaviest 5 bps drag; quarterly rebalancing minimises drawdown but loses tracking precision. The constraint sweep makes the trade-off explicit.
  \item \textbf{Long rolling window beats recent adaptation.} The 3-year window dominates 1-year on net IR for every linear model --- a stable signal beats reactive fitting on weekly data.
  \item \textbf{Predict-then-optimize closes the constraint gap.} The convex layer applies the $L_1$-gross and per-asset $|w_j|$ caps inside the optimisation; Ridge with post-hoc rescaling pays a small TE premium for the same risk profile.
  \item \textbf{Kalman + EM matches the static $\sigma_w = 10^{-3}$ grid point.} EM is a useful sanity check; the regime-switching variant narrows TE in 2008 at the cost of more turnover.
  \item \textbf{The MLP rivals the linear leaderboard} when warm-started from Ridge and penalised on turnover; without $\lambda \lVert \Delta w_t \rVert_1$, NN variants over-trade and lose net IR.
\end{itemize}
\end{frame}

% =============================================================
\begin{frame}{Recommended configuration}
\textbf{Deployable choice (composite-score winner).}
\begin{itemize}
  \item \textbf{Model:} PO + ElasticNet on top-5 futures (pre-selected by training-window correlation).
  \item \textbf{Cadence:} $\Delta t = 4$ weeks --- monthly rebalancing balances tracking and turnover.
  \item \textbf{Risk:} gross-exposure cap $L = 1.5$, iterative VaR cap $\le 8\%$.
  \item \textbf{Transaction cost:} 5 bps on L1 turnover; budget $\tau \in \{2, 5, 10\}$ bps stress-tested.
\end{itemize}

\smallskip
\textbf{Why this configuration.}
\begin{itemize}
  \item \textbf{Net IR survives the cost sweep.} Monthly cadence keeps $\bar T$ low; top-5 prunes redundant exposures.
  \item \textbf{Crisis robustness.} Composite ranking weights MDD and TE alongside IR; the winner is consistent across the 2008 and 2020 windows.
  \item \textbf{Interpretability.} The linear coefficients are portfolio weights --- the strategy is transparent for a regulatory audit.
\end{itemize}
\end{frame}

% =============================================================
\begin{frame}{Takeaways}
\begin{itemize}
  \item \textbf{Methodology before models.} The Part II harness fixes rolling-window discipline, transaction-cost accounting, and the VaR / GE projection layer \emph{once}; every Part III--VI model plugs into the same evaluation surface.
  \item \textbf{Leakage discipline runs end-to-end.} Top-$K$ pre-selection inside the training window, HMM frozen on the warm-up slice, train-fold standardisation for the NN, $\texttt{shift(1)}$ on every lagged feature --- every fit-stage is closed against future data.
  \item \textbf{Turnover discipline is non-optional.} Without explicit penalisation, every adaptive model (NN, regime-switching Kalman) over-trades and loses on net IR.
  \item \textbf{Constraint sweep is the deployment lens.} Cadence and leverage choices dominate model choice once costs and risk caps are in play; the composite score makes the trade-off auditable.
  \item \textbf{Single pickle contract.} \texttt{ReplicaResult} flows through every track unchanged --- that is what lets Part VII be a one-page comparison rather than five fragmented narratives.
\end{itemize}
