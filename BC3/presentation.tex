	\textbf{Goal.} Replicate a hidden ``Monster Index'' using only liquid Bloomberg futures, subject to a regulatory VaR cap.
	\[
		r^{\text{target}}_t = 0.50\,r^{\text{HFRX}}_t + 0.25\,r^{\text{MSCI World}}_t + 0.25\,r^{\text{Global Bond}}_t
	\]

	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\textbf{Universe (11 futures).}
			\begin{itemize}
				\item Rates: RX1, TY1, DU1, TU2
				\item Equity: ES1, NQ1, VG1, TP1, LLL1
				\item Commodity: GC1, CO1
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{Constraints.}
			\begin{itemize}
				\item Sample: weekly, Oct 2007 -- Apr 2021 ($704$ obs).
				\item Gaussian VaR cap (1\%, 1-month) $\le 8\%$.
				\item Gross-exposure cap: $\sum_j \lvert w_j \rvert \le L$.
				\item Transaction cost: $\tau = 5$ bps on L1 turnover.
			\end{itemize}
		\end{column}
	\end{columns}

\begin{frame}{Backtest methodology --- Part I}
	\textbf{Rolling walk-forward.} At every rebalance week $t$:
	\begin{enumerate}
		\item \textbf{Fit} on $X[t{-}104:t],\, y[t{-}104:t]$ (no look-ahead --- $X[t],\, y[t]$ never seen).
		\item \textbf{Project} onto the feasible set: gross cap $\sum_j |w_j| \le L$, then iterative shrinkage until $\mathrm{VaR}(w) \le 8\%$.
		\item \textbf{Trade} $\Delta w_t = w^{\text{new}} - w^{\text{old}}$; \textbf{cost} $c_t = \tau \cdot \sum_j |\Delta w_{t,j}|$.
		\item \textbf{Realise} $r^{\text{rep}}_t = w_t^\top r_t$; on non-rebalance weeks carry $w_{t-1}$ forward.
	\end{enumerate}

	\smallskip
	\textbf{Contract metrics.}
	\[
		\mathrm{TE} = \sqrt{52\,\mathrm{Var}(r^{\text{rep}}_t - y_t)}, \quad
		\mathrm{IR} = \frac{52\,\mathbb{E}[r^{\text{rep}}_t - y_t]}{\mathrm{TE}}, \quad
		\mathrm{net\_IR} = \frac{52\,\mathbb{E}[r^{\text{rep}}_t - c_t - y_t]}{\mathrm{net\_TE}}
	\]

	\smallskip
	\textbf{VaR fallback.} Jarque-Bera on the target rejects normality ($p \approx 0$); Gaussian VaR (4.14\%) understates the historical 1\%-quantile (5.96\%) --- the harness can fall back to a historical VaR projector.
\end{frame}

\begin{frame}{Part III --- predict-then-optimize (convex layer)}
	\textbf{Two-stage rebalance.}
	\begin{enumerate}
		\item \textbf{Alpha.} Fit a classical linear model on $(X_{\text{tr}}, y_{\text{tr}})$; take $\mu = \texttt{coef\_}$.
		\item \textbf{Risk.} Ledoit--Wolf shrunken sample covariance $\Sigma$ (annualised).
		\item \textbf{Optimise} (CVXPY):
	\end{enumerate}
	\[
		\max_{w} \;\; \mu^{\top} w \;-\; \tfrac{1}{2}\lambda\, w^{\top}\Sigma\, w \;-\; \tau\,\lVert w - w_{t-1}\rVert_1
		\quad \text{s.t.} \quad \lVert w \rVert_1 \le L,\;\; \lvert w_j \rvert \le 0.5
	\]

	\smallskip
	\textbf{Sanity check.} With $\lambda = 2$, slack constraints, no LW shrinkage, the convex layer recovers OLS to within solver tolerance.

	\smallskip
	\textbf{Findings.}
	\begin{itemize}
		\item Longest rolling window (3y) and modest Ridge dominate net IR on the leaderboard.
		\item Huber is competitive only in the 2008/2020 windows.
		\item Lasso's sparsity buys lower turnover but loses on TE.
	\end{itemize}
\end{frame}

\begin{frame}{Part IV --- Kalman filter / state-space formulation}
	{\footnotesize The state-space model treats portfolio weights as a latent random walk observed through realised returns.}
	\[
		\begin{aligned}
			x_t & = A\, x_{t-1} + B\, u_t, \quad         & u_t           & \sim \mathcal{N}(0, I_{11}) &  & \text{(state: portfolio weights, random walk)} \\
			y_t & = C_t\, x_t + D\, \varepsilon_t, \quad & \varepsilon_t & \sim \mathcal{N}(0, 1)      &  & \text{(observation: target return)}
		\end{aligned}
	\]
	{\footnotesize $A = I_{11}$,\; $B = \sigma_w \cdot I_{11}$,\; $C_t = r_t^\top$,\; $D = \hat{\sigma}_y$ from the training window,\;
		$x_t \in \mathbb{R}^{11}$ the latent weight vector,\; $y_t$ the scalar target return.
		Ridge regression warm-start on the first 52 weeks; filter implemented from scratch.}

	\vspace{2pt}
	\textbf{Three variants:}
	\begin{columns}[T]
		\begin{column}{0.50\textwidth}
			\footnotesize
			\begin{itemize}\setlength{\itemsep}{1pt}
				\item \textbf{Basic Kalman} --- fixed process noise $\sigma_w = 0.001$, fixed $R$, with a VaR guardrail.
				\item \textbf{Tuned Kalman} --- grid search over $(Q, R_{\text{scale}}, P_0)$ with a Universal Replication Score, plus Q-sensitivity analysis.
				\item \textbf{Double Adaptive Volatility} --- both $Q$ and $R$ scale dynamically with rolling volatility.
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\footnotesize
			\textbf{Parameter Robustness.} $\pm20\%$ perturbation grid confirms structural soundness.

			\vspace{2pt}
			\textbf{Look-ahead discipline.} $\hat{x}_{t|t}$ uses $y_t$ (unavailable at the open of week $t$). We shift weights by one week: held weight at $t$ depends only on observations through $t{-}1$.
		\end{column}
	\end{columns}
\end{frame}

% =============================================================
\begin{frame}{Part IV --- Kalman constraints \& optimisation score}
	\begin{columns}[T]
		\begin{column}{0.50\textwidth}
			\textbf{1.\ Turnover Dampening}
			\[
				\hat{w}_t^{\text{damp}} = (1 - \alpha)\,\hat{w}_{t-1}^{\text{final}} + \alpha\, \hat{x}_{t|t}
			\]
			{\scriptsize Exponential smoothing on raw Kalman updates, structurally slowing weight adjustments to prevent excessive trading and reduce transaction costs.}

			\smallskip
			\textbf{2.\ Gross Exposure Cap}
			{\scriptsize
				\[
					\hat{w}_t^{\text{gross}} =
					\begin{cases}
						\hat{w}_t^{\text{damp}} \dfrac{GE_{\text{cap}}}{\sum_i |\hat{w}_{i,t}^{\text{damp}}|}
						                        & \sum_i |\hat{w}_{i,t}^{\text{damp}}| > GE_{\text{cap}} \\[4pt]
						\hat{w}_t^{\text{damp}} & \text{otherwise}
					\end{cases}
				\]
			}
			{\scriptsize Proportional scale-down if the absolute sum of weights exceeds $GE_{\text{cap}} = 2.0$.}
		\end{column}
		\begin{column}{0.50\textwidth}
			\textbf{3.\ VaR Guardrail}
			{\footnotesize
				\[
					\hat{w}_t^{\text{final}} =
					\begin{cases}
						\hat{w}_t^{\text{gross}} \dfrac{VaR_{\text{cap}}}{\widehat{VaR}} & \widehat{VaR} > 8\% \\[4pt]
						\hat{w}_t^{\text{gross}}                                         & \text{otherwise}
					\end{cases}
				\]
			}{\scriptsize where $\widehat{VaR} = VaR_{1\%,\,4W}(\hat{w}_t^{\text{gross}})$.
				Aggressively downweights if forward risk breaches $8\%$.}

			\smallskip
			\textbf{4.\ Replication Score}
			{\footnotesize
				\[
					S = \rho(\mathbf{r}_R, \mathbf{r}_T) - 2\,TE_n - 0.5\,\overline{\Delta w} - 0.1\,MDD_n
				\]
			}{\scriptsize $TE_n$ = net TE, $\overline{\Delta w}$ = avg turnover, $MDD_n$ = net max drawdown.}
		\end{column}
	\end{columns}
\end{frame}
