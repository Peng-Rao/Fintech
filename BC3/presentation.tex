\begin{frame}{Problem and data}
	\textbf{Goal.} Replicate a hidden ``Monster Index'' using only liquid Bloomberg futures, subject to a regulatory VaR cap.
	\[
		r^{\text{target}}_t = 0.50\,r^{\text{HFRX}}_t + 0.25\,r^{\text{MSCI World}}_t + 0.25\,r^{\text{Global Bond}}_t
	\]

	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\textbf{Universe (11 futures).}
			\begin{itemize}
				\item Rates: RX1, TY1, DU1, TU2
				\item Equity: ES1, NQ1, VG1, TP1, LLL1
				\item Commodity: GC1, CO1
			\end{itemize}
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{Constraints.}
			\begin{itemize}
				\item Sample: weekly, Oct 2007 -- Apr 2021 ($704$ obs).
				\item Gaussian VaR cap (1\%, 1-month) $\le 8\%$.
				\item Gross-exposure cap: $\sum_j \lvert w_j \rvert \le L$.
				\item Transaction cost: $\tau = 5$ bps on L1 turnover.
			\end{itemize}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}{Backtest methodology --- Part I}
	\textbf{Rolling walk-forward.} At every rebalance week $t$:
	\begin{enumerate}
		\item \textbf{Fit} on $X[t{-}104:t],\, y[t{-}104:t]$ (no look-ahead --- $X[t],\, y[t]$ never seen).
		\item \textbf{Project} onto the feasible set: gross cap $\sum_j |w_j| \le L$, then iterative shrinkage until $\mathrm{VaR}(w) \le 8\%$.
		\item \textbf{Trade} $\Delta w_t = w^{\text{new}} - w^{\text{old}}$; \textbf{cost} $c_t = \tau \cdot \sum_j |\Delta w_{t,j}|$.
		\item \textbf{Realise} $r^{\text{rep}}_t = w_t^\top r_t$; on non-rebalance weeks carry $w_{t-1}$ forward.
	\end{enumerate}

	\smallskip
	\textbf{Contract metrics.}
	\[
		\mathrm{TE} = \sqrt{52\,\mathrm{Var}(r^{\text{rep}}_t - y_t)}, \quad
		\mathrm{IR} = \frac{52\,\mathbb{E}[r^{\text{rep}}_t - y_t]}{\mathrm{TE}}, \quad
		\mathrm{net\_IR} = \frac{52\,\mathbb{E}[r^{\text{rep}}_t - c_t - y_t]}{\mathrm{net\_TE}}
	\]

	\smallskip
	\textbf{VaR fallback.} Jarque-Bera on the target rejects normality ($p \approx 0$); Gaussian VaR (4.14\%) understates the historical 1\%-quantile (5.96\%) --- the harness can fall back to a historical VaR projector.
\end{frame}

\begin{frame}{Part III --- predict-then-optimize (convex layer)}
	\textbf{Two-stage rebalance.}
	\begin{enumerate}
		\item \textbf{Alpha.} Fit a classical linear model on $(X_{\text{tr}}, y_{\text{tr}})$; take $\mu = \texttt{coef\_}$.
		\item \textbf{Risk.} Ledoit--Wolf shrunken sample covariance $\Sigma$ (annualised).
		\item \textbf{Optimise} (CVXPY):
	\end{enumerate}
	\[
		\max_{w} \;\; \mu^{\top} w \;-\; \tfrac{1}{2}\lambda\, w^{\top}\Sigma\, w \;-\; \tau\,\lVert w - w_{t-1}\rVert_1
		\quad \text{s.t.} \quad \lVert w \rVert_1 \le L,\;\; \lvert w_j \rvert \le 0.5
	\]

	\smallskip
	\textbf{Sanity check.} With $\lambda = 2$, slack constraints, no LW shrinkage, the convex layer recovers OLS to within solver tolerance.

	\smallskip
	\textbf{Findings.}
	\begin{itemize}
		\item Longest rolling window (3y) and modest Ridge dominate net IR on the leaderboard.
		\item Huber is competitive only in the 2008/2020 windows.
		\item Lasso's sparsity buys lower turnover but loses on TE.
	\end{itemize}
\end{frame}

\begin{frame}{Part IV --- Kalman filter / state-space}
	\begin{columns}[T]
		\begin{column}{0.50\textwidth}
			\textbf{Random-walk weights model.}
			\[
				\begin{aligned}
					x_t & = x_{t-1} + B\, u_t,                  & u_t           & \sim \mathcal{N}(0, I_{11}) \\
					y_t & = r_t^{\top} x_t + D\, \varepsilon_t, & \varepsilon_t & \sim \mathcal{N}(0, 1)
				\end{aligned}
			\]
			$B = \sigma_w I$, $D = \hat\sigma_y$; Ridge warm-start on the first 52 weeks.
			\medskip

			\textbf{Variants:}
			\begin{itemize}
				\item Static $\sigma_w$ grid $\{10^{-4}, 10^{-3}, 10^{-2}\}$.
				\item EM-tuned $(B, D)$, capped at 3 iterations.
				\item Double-adaptive: $\sigma_w$ and $D$ rescale with rolling vol.
			\end{itemize}

			\textbf{Leakage rule.} Filter outputs $\hat x_{t|t}$ use $y_t$, so we \emph{shift weights by one week} before evaluation.
		\end{column}
		\begin{column}{0.50\textwidth}
			\centering
			\includegraphics[width=\linewidth]{p4_kalman_adaptive_replication.png}\\[-2pt]
			{\scriptsize Adaptive Kalman replica vs target.}
		\end{column}
	\end{columns}
\end{frame}
